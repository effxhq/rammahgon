Download the CLI binary and add to your `PATH` with the following commands:

<Tabs
  defaultValue='darwin'
  values={[

  <TabItem value='linux32'>
    <CodeBlock className='language-shell'>
      curl -Lo effx
      https://effx.run/effx-cli/releases/latest/effx-cli_Linux_i386{'\n'}
      sudo install effx /usr/local/bin/
    </CodeBlock>
  </TabItem>
  <TabItem value='linux64'>
    <CodeBlock className='language-shell'>
      curl -Lo effx
      https://effx.run/effx-cli/releases/latest/effx-cli_Linux_x86_64{'\n'}
      sudo install effx /usr/local/bin/
    </CodeBlock>
  </TabItem>
  <TabItem value='darwin'>

  </TabItem>

    </CodeBlock>
  </TabItem>
</Tabs>
